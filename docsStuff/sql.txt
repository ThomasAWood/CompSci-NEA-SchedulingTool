CREATE TABLE `booking` (
  `id` int NOT NULL AUTO_INCREMENT,
  `studentId` int NOT NULL,
  `lessonId` int NOT NULL,
  `date` datetime NOT NULL,
  `cancelled` tinyint NOT NULL,
  PRIMARY KEY (`id`),
  CONSTRAINT `lessonId` FOREIGN KEY (`lessonId`) REFERENCES `lessons` (`id`),
  CONSTRAINT `studentId` FOREIGN KEY (`studentId`) REFERENCES `users` (`id`)
)

CREATE TABLE `lessons` (
  `id` int NOT NULL AUTO_INCREMENT,
  `teacherId` int NOT NULL,
  `startDateTime` datetime NOT NULL,
  `duration` varchar(10) NOT NULL,
  `endDateTime` datetime DEFAULT NULL,
  `isRecurring` tinyint NOT NULL,
  PRIMARY KEY (`id`),
  CONSTRAINT `teacherId` FOREIGN KEY (`teacherId`) REFERENCES `users` (`id`)
)

CREATE TABLE `users` (
  `id` int NOT NULL AUTO_INCREMENT,
  `email` varchar(255) NOT NULL,
  `password` varchar(32) NOT NULL,
  `create_time` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  `fname` varchar(45) NOT NULL,
  `lname` varchar(45) NOT NULL,
  `isTeacher` tinyint NOT NULL,
  `hourly` int DEFAULT NULL,
  PRIMARY KEY (`id`)
)
